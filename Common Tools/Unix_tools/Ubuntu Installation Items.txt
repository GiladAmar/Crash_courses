--  Set swappiness:
        See current swapfile size: sudo swapon -s
        sudo swapoff -a
        sudo fallocate -l 8G /swapfile # recommended between 1 and 2x your RAM
        sudo chmod 600 /swapfile
        sudo mkswap /swapfile
        sudo swapon /swapfile
        sudo swapon -s
        echo '/swapfile none swap sw 0 0' | sudo tee -a /etc/fstab
